                                                                          # 1.实现一个LazyMan,可以按照以下方式调用

                                                                          解决方案

                                                                          class _LazyMan{
  constructor(name){
    //异步执行队列
    this.listeners = [];
    //初始化事件
    this.listeners.push(()=>{
      console.log(`Hi This is ${name}`);
    });

    queueMicrotask(()=>{
      this.run();
    });
  }

  run(){
    let {listeners} = this,
        index=0;
      const next=async ()=>{
        if(index>listeners.length-1) return;
        const task=listeners[index++];
        await task();
        next();
      }
      next();
  }

  eat(food){
    this.listeners.push(()=>{
      console.log(`Eat ${food}`);
    });
    return this;
  }

  sleep(time,flag=false){
    if(flag) {
      this.listeners.unshift(()=>{
        setTimeout(()=>{
          console.log(`Weap up at ${time}`);
        },time);
      })
      return this;
    }
    this.listeners.push(()=>{
      setTimeout(()=>{
        console.log(`Weap up at ${time}`);
      },time);
    });
    return this;
  }

  sleepFirst(time){
    return sleep(time,true);
  }
}

function LazyMan(name){
  return new _LazyMan(name);
}

LazyMan("Hank").sleep(10).eat('dinner')

                                                                          # 2.有ajax的异步并发管控

                                                                          题目

                                                                          现有一个POST接口：http://xxx.com/students,每次请求只能返回10个学生的课程成绩，如下

                                                                          [

 {name:'张三',score:99,time:'2021-12-22'},

 {name:'李四',score:99,time:'2021-12-12'},

 ...

]

                                                                          该接口有一定概率请求失败，不可忽略

                                                                          要求，实现一个函数，总共获取100个成绩大于90，且时间在2021年12月03日之后的学生的课程成绩，并按从小到大顺序排序，请求用axios

                                                                          提示：

                                                                          浏览器最多有6个并行的网络请求

                                                                          //模拟数据请求的方法
const source=axios.CancelToken.source();
const query=function query(){
  return axios.post('https://localhost:8080/students',null,{cancelToken:source.token}).then(response=>response.data)
}

const fetchStudents=function fetchStudents(){
  return new Promise(function(resolve, reject){
    //同时开启六个工作区
    let works=new Array(6).fill(null),
        values=[];
    works.forEach(function(){
      async function next(){
        if(values.length>=100){
          //请求结束:取消还在发送的请求
          source.cancel();
          resolve(values.slice(0,100).sort((a,b)=>b.score-a.score));
          return;
        }
        try{
          let value=await query();
          value=value.filter(item=>(item.score>90 && (new Date(item.time)-new Date('2012-12-03')>0)));
          values=values.concat(value);
        }catch(_){}
      }
      next();
    })
  })
}

                                                                          # 并发控制简介

                                                                          你可能会遇到并发控制的场景，比如控制请求并发数。那么在 JavaScript 中如何实现并发控制呢？在回答这个问题之前，我们来简单介绍一下并发控制。

                                                                          假设有 6 个待办任务要执行，而我们希望限制同时执行的任务个数，即最多只有 2 个任务能同时执行。当 正在执行任务列表 中的任何 1 个任务完成后，程序会自动从 待办任务列表 中获取新的待办任务并把该任务添加到 正在执行任务列表 中。为了让大家能够更直观地理解上述的过程，阿宝哥特意画了以下 3 张图：

class TaskQueue {
  constructor(tasks, limit, onComplete) {
    let self = this;
    self.tasks = tasks;
    self.limit = limit;
    self.onComplete = onComplete;
    //正在运行的任务数量
    self.running = 0;
    //当前取出任务的索引
    self.index = 0;
    //结果
    self.values = [];
    for(let i=0;i<limit;i++){
        self.next();
    }
  }

  pushStack(task) {
    this.tasks.push(task);
    this.next();
  }

  async next() {
    let self = this;
    let result;
    // let running=this.running;
    let { running, limit, index, values, tasks, onComplete } = self;
    // console.log(running)
    // console.log(this.running, limit, index, values, tasks, onComplete);
    if (running < limit && index <= tasks.length - 1) {
      //没运行满
      self.running++;
      let curIndex = index;
      let task = tasks[self.index++];
      try {
        result = await task();
        values[curIndex] = result;
      } catch (e) {
        values[curIndex] = null;
      }
      self.running--;
      console.log('self',self,self.index ,tasks.length ,self.running)
      if (self.index == tasks.length && self.running == 0) {
        //任务都运行完了
        console.log('ok!',onComplete,values)
        onComplete(values);
      }
      self.next();
    }
  }
}

const createRequest = function createRequest(tasks, limit, onComplete) {
  if (!Array.isArray(tasks))
    throw new TypeError("tasks must be an array of tasks");
  if (typeof limit === "function") onComplete = limit;
  limit = +limit;
  if (isNaN(limit)) limit = 2;
  //Function.prototype => ()=>{} 一个空函数
  if (typeof onComplete !== "function") onComplete = Function.prototype;
  
  return new TaskQueue(tasks, limit, onComplete);
};
